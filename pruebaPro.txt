public class fizzbuzz {
    public static void main(String[] args) {
        for(int i=1; i<=100; i++){
            if(i%3 == 0 && i%5 == 0){
                System.out.println("Numero " + i +": fizzbuzz");
            }else if(i%3 == 0){
                System.out.println("Numero " + i +": fizz");
            }else if(i%5 == 0){
                System.out.println("Numero " + i +": buzz");
            }   
        }
    }
}
